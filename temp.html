Program 1: Array Basic Operations, Indexing and Slicing

import numpy as np

arr = np.array([10,20,30,40,50])

print("Original:", arr)
print("Add 5:", arr+5)
print("Multiply by 2:", arr*2)
print("Sum:", arr.sum())

print("First Element:", arr[0])
print("Last Element:", arr[-1])

print("Index 1 to 3:", arr[1:4])
print("Start to 2:", arr[:3])
print("2 to End:", arr[2:])

Program 2: Array Aliasing and Copying

import numpy as np

arr = np.array([10,20,30,40])

alias_arr = arr
copy_arr = arr.copy()

arr[0] = 100

print("Original array (after change):", arr)
print("Aliased array (changes reflect):", alias_arr)
print("Copied array (no change):", copy_arr)
